Automated Supplier Quote Request System
Business Context
Renan now has a great quote comparison system (Task 2), but he's still manually requesting quotes from suppliers via WhatsApp/Email. This is slow and inconsistent.

Current Pain Points:

Time-consuming: Renan spends 15-20 minutes per client crafting RFO emails

Inconsistent format: Each supplier wants data in different formats

No tracking: Hard to know which suppliers responded, which didn't

Manual follow-up: Renan has to remember to chase suppliers

Error-prone: Typos in consumption data lead to wrong quotes

Goal: Build an "RFO Automático" system where Renan can:

One-click send standardized RFOs to multiple suppliers

Track responses automatically

Auto-import supplier quotes into the system

Send reminders to non-responsive suppliers

Maintain professional consistency across all requests

🏗️ ARCHITECTURE OVERVIEW
Core Concept:
text
Client Data (from Task 1) 
    → RFO Generator (Task 3) 
    → Send to Suppliers (Email/API) 
    → Track Responses 
    → Auto-import into Quotes (Task 2)
Three Send Methods:
Email (MVP): Send structured RFO via email to supplier sales teams

Supplier Portal (Future): Auto-fill supplier web forms

API (Ideal Future): Direct integration with supplier systems

CREATE TABLE rfo_requests (
  id SERIAL PRIMARY KEY,
  
  -- Relationships
  client_id INTEGER REFERENCES clients(id) NOT NULL,
  bill_upload_id INTEGER REFERENCES bill_uploads(id), -- Specific bill analyzed
  
  -- RFO Metadata
  rfo_number VARCHAR(50) UNIQUE NOT NULL, -- Format: RFO-2024-001
  rfo_date TIMESTAMP DEFAULT NOW(),
  status VARCHAR(20) DEFAULT 'draft', -- 'draft', 'sent', 'partial_response', 'complete', 'expired'
  
  -- Request Details
  requested_suppliers TEXT[], -- Array of supplier IDs/names
  response_deadline DATE NOT NULL,
  priority VARCHAR(20) DEFAULT 'normal', -- 'low', 'normal', 'high', 'urgent'
  
  -- Client Data Snapshot (frozen at time of RFO)
  snapshot_consumption_kwh DECIMAL(10,2),
  snapshot_demanda_kw DECIMAL(10,2),
  snapshot_uc VARCHAR(50),
  snapshot_distribuidora VARCHAR(100),
  snapshot_contract_end DATE,
  
  -- Tracking
  sent_count INTEGER DEFAULT 0, -- How many suppliers received it
  response_count INTEGER DEFAULT 0, -- How many responded
  last_sent_date TIMESTAMP,
  
  -- Attachments/Content
  email_subject VARCHAR(200),
  email_body TEXT,
  attachments JSONB, -- Array of attachment metadata
  
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  
  -- Constraints
  CONSTRAINT valid_deadline CHECK (response_deadline > CURRENT_DATE)
);

-- Indexes
CREATE INDEX idx_rfo_requests_client_id ON rfo_requests(client_id);
CREATE INDEX idx_rfo_requests_status ON rfo_requests(status);
CREATE INDEX idx_rfo_requests_rfo_number ON rfo_requests(rfo_number);
CREATE UNIQUE INDEX idx_rfo_requests_number_unique ON rfo_requests(rfo_number);

import React, { useState, useEffect } from 'react';
import './RFOTrackerDashboard.css';

const RFOTrackerDashboard = () => {
  const [rfos, setRfos] = useState([]);
  const [filter, setFilter] = useState('all'); // 'all', 'pending', 'responded', 'expired'
  const [selectedRFO, setSelectedRFO] = useState(null);

  useEffect(() => {
    fetchRFOs();
  }, [filter]);

  const fetchRFOs = async () => {
    const response = await fetch(`/api/rfos?filter=${filter}`);
    const data = await response.json();
    setRfos(data);
  };

  const calculateResponseRate = (rfo) => {
    if (!rfo.sent_count) return 0;
    return Math.round((rfo.response_count / rfo.sent_count) * 100);
  };

  const getDaysRemaining = (deadline) => {
    const today = new Date();
    const deadlineDate = new Date(deadline);
    const diffTime = deadlineDate - today;
    const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
    return diffDays;
  };

  const sendReminder = async (rfoId, supplierId) => {
    await fetch(`/api/rfos/${rfoId}/reminder/${supplierId}`, {
      method: 'POST'
    });
    alert('Lembrete enviado!');
    fetchRFOs();
  };

  const markAsResponded = async (trackingId, quoteData) => {
    await fetch(`/api/rfo-tracking/${trackingId}/responded`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(quoteData)
    });
    alert('Resposta registrada!');
    fetchRFOs();
  };

  return (
    <div className="rfo-tracker">
      {/* HEADER */}
      <div className="tracker-header">
        <h3>📊 Acompanhamento de RFOs</h3>
        <div className="tracker-stats">
          <div className="tracker-stat">
            <div className="stat-number">{rfos.length}</div>
            <div className="stat-label">RFOs Ativos</div>
          </div>
          <div className="tracker-stat">
            <div className="stat-number">
              {rfos.reduce((sum, rfo) => sum + rfo.response_count, 0)}
            </div>
            <div className="stat-label">Respostas</div>
          </div>
          <div className="tracker-stat">
            <div className="stat-number">
              {rfos.filter(rfo => getDaysRemaining(rfo.response_deadline) < 0).length}
            </div>
            <div className="stat-label">Vencidos</div>
          </div>
          <div className="tracker-stat">
            <div className="stat-number">
              {rfos.filter(rfo => rfo.status === 'complete').length}
            </div>
            <div className="stat-label">Completos</div>
          </div>
        </div>
      </div>

      {/* FILTERS */}
      <div className="filter-bar">
        <button 
          className={filter === 'all' ? 'active' : ''}
          onClick={() => setFilter('all')}
        >
          Todos
        </button>
        <button 
          className={filter === 'pending' ? 'active' : ''}
          onClick={() => setFilter('pending')}
        >
          Pendentes
        </button>
        <button 
          className={filter === 'partial' ? 'active' : ''}
          onClick={() => setFilter('partial')}
        >
          Parciais
        </button>
        <button 
          className={filter === 'complete' ? 'active' : ''}
          onClick={() => setFilter('complete')}
        >
          Completos
        </button>
        <button 
          className={filter === 'expired' ? 'active' : ''}
          onClick={() => setFilter('expired')}

// POST /api/clients/:id/rfo/send
router.post('/:id/rfo/send', async (req, res) => {
  try {
    const clientId = req.params.id;
    const { rfoNumber, suppliers, emailContent, deadline, priority, customEmail } = req.body;
    
    // Get client data
    const client = await db.selectFrom('clients')
      .select(['company_name', 'avg_consumption_kwh', 'subgrupo'])
      .where('id', '=', clientId)
      .executeTakeFirst();
    
    // Get latest bill data
    const latestBill = await db.selectFrom('bill_uploads')
      .select(['uc_code', 'demanda_kw', 'distribuidora'])
      .where('client_id', '=', clientId)
      .orderBy('created_at', 'desc')
      .limit(1)
      .executeTakeFirst();
    
    // Create RFO record
    const rfoRecord = await db.insertInto('rfo_requests').values({
      client_id: clientId,
      rfo_number: rfoNumber,
      status: 'sent',
      requested_suppliers: suppliers.map(s => s.name),
      response_deadline: deadline,
      priority: priority,
      snapshot_consumption_kwh: client.avg_consumption_kwh,
      snapshot_demanda_kw: latestBill?.demanda_kw,
      snapshot_uc: latestBill?.uc_code,
      snapshot_distribuidora: latestBill?.distribuidora,
      email_subject: emailContent.subject,
      email_body: emailContent.body,
      sent_count: suppliers.length,
      response_count: 0
    }).returning('id').execute();
    
    const rfoId = rfoRecord[0].id;
    
    // Send to each supplier
    let sentCount = 0;
    const trackingRecords = [];
    
    for (const supplier of suppliers) {
      try {
        // Get primary contact for supplier
        const contact = supplier.contacts.find(c => c.is_primary) || supplier.contacts[0];
        
        if (!contact?.email) continue;
        
        // Send email (using Nodemailer)
        const emailResult = await sendRFOMail({
          to: contact.email,
          subject: emailContent.subject,
          body: emailContent.body,
          attachments: emailContent.attachments,
          replyTo: customEmail || 'renan@otimaenergia.com.br'
        });
        
        // Create tracking record
        const tracking = await db.insertInto('rfo_supplier_tracking').values({
          rfo_id: rfoId,
          supplier_id: supplier.id,
          contact_name: contact.name,
          contact_email: contact.email,
          sent_status: emailResult.success ? 'sent' : 'failed',
          sent_date: new Date(),
          sent_method: 'email',
          error_message: emailResult.error || null
        }).returning('id').execute();
        
        trackingRecords.push(tracking[0]);
        sentCount++;
        
      } catch (supplierError) {
        console.error(`Error sending to ${supplier.name}:`, supplierError);
        
        // Still create tracking record for failed attempts
        await db.insertInto('rfo_supplier_tracking').values({
          rfo_id: rfoId,
          supplier_id: supplier.id,
          sent_status: 'failed',
          error_message: supplierError.message
        }).execute();
      }
    }
    
    // Update RFO with actual sent count
    await db.updateTable('rfo_requests')
      .set({ sent_count: sentCount })
      .where('id', '=', rfoId)
      .execute();
    
    // Return success with tracking info
    res.json({
      success: true,
      rfo: {
        id: rfoId,
        rfo_number: rfoNumber,
        sent_count: sentCount,
        response_deadline: deadline,
        status: 'sent'
      },
      tracking: trackingRecords
    });
    
  } catch (error) {
    console.error('RFO send error:', error);
    res.status(500).json({
      success: false,
      error: 'Erro ao enviar RFO. Por favor, envie manualmente.'
    });
  }
});

// Helper: Send email via Nodemailer
async function sendRFOMail({ to, subject, body, attachments, replyTo }) {
  const transporter = nodemailer.createTransport({
    host: process.env.SMTP_HOST,
    port: process.env.SMTP_PORT,
    secure: true,
    auth: {
      user: process.env.SMTP_USER,
      pass: process.env.SMTP_PASS
    }
  });
  
  const mailOptions = {
    from: `"Ótima Energia" <${process.env.SMTP_FROM}>`,
    to: to,
    replyTo: replyTo,
    subject: subject,
    text: body,
    html: convertToHTML(body), // Convert plain text to basic HTML
    attachments: attachments.map(att => ({
      filename: att.name,
      path: `/attachments/${att.name}`, // Path to stored files
      contentType: getContentType(att.type)
    }))
  };
  
  try {
    const info = await transporter.sendMail(mailOptions);
    
    // Track email opens (pixel tracking)
    const trackingPixel = `<img src="${process.env.APP_URL}/api/track/open/${info.messageId}" width="1" height="1" />`;
    
    return {
      success: true,
      messageId: info.messageId
    };
    
  } catch (error) {
    return {
      success: false,
      error: error.message
    };
  }
}
