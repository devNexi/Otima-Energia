typescriptapp.post("/api/admin/seed-conference-deals", async (req, res) => {
  if (!await validateDealOsSession(req, res)) return;
  try {
    const { deals, dealQuotes, dealTracks, clients: clientsTable } = await import("@shared/schema");
    const { db } = await import("./db");
    const { eq, ne, isNotNull } = await import("drizzle-orm");

    // Get REAL suppliers from the database
    const allSuppliers = await db.select().from(suppliersTable).where(eq(suppliersTable.isActive, true));
    if (allSuppliers.length < 3) {
      return res.status(400).json({ success: false, error: 'Precisa de pelo menos 3 fornecedores. Execute "Importar Fornecedores" primeiro.' });
    }

    // Build a name → id map
    const supplierMap: Record<string, number> = {};
    for (const s of allSuppliers) {
      supplierMap[s.name] = s.id;
    }

    const primeId = supplierMap['Prime Energy'];
    const atmoId = supplierMap['ATMO Energia'];
    const cemigId = supplierMap['CEMIG'];
    const delantisId = supplierMap['Delantis Energias Renováveis'];
    const genialId = supplierMap['Genial Energia'];

    // Pick 4 REAL existing clients (ones with CNPJs, not demo)
    const existingClients = await db.select().from(clientsTable)
      .where(isNotNull(clientsTable.cnpj))
      .limit(4);

    if (existingClients.length < 2) {
      return res.status(400).json({ success: false, error: 'Precisa de pelo menos 2 clientes cadastrados com CNPJ.' });
    }

    // Check if conference deals already exist
    const existingConf = await db.select().from(deals).where(eq(deals.isDemo, true)).limit(1);
    if (existingConf.length > 0) {
      return res.json({ success: true, message: 'Deals de conferência já existem.', skipped: true });
    }

    // Conference deals — realistic Brazilian energy pricing
    // GD BT prices: R$270-330/MWh range (discount off distributor tariff)
    // ACL Convencional: R$220-280/MWh range
    // ACL Incentivada 50%: R$200-250/MWh range
    const conferenceDeals = [
      {
        // GD deal — client on Light RJ (Genial + Prime territory)
        clientId: existingClients[0].id,
        trackType: 'GDL',
        energyType: 'gd_assinatura',
        submarket: 'SE_CO',
        status: 'QUOTES_RECEIVED',
        quotes: [
          { supplierId: primeId, name: 'Prime Energy', price: 285.50, term: 24, structure: 'GD Assinatura BT', responseDays: 1 },
          { supplierId: genialId, name: 'Genial Energia', price: 278.00, term: 24, structure: 'GD Assinatura BT', responseDays: 2 },
          { supplierId: atmoId, name: 'ATMO Energia', price: 292.00, term: 12, structure: 'GD Assinatura BT', responseDays: 3 },
          { supplierId: delantisId, name: 'Delantis Energias Renováveis', price: 305.00, term: 24, structure: 'GD Compensação Nacional', responseDays: 2 },
          { supplierId: primeId, name: 'Prime Energy', price: 272.00, term: 36, structure: 'GD Assinatura BT', responseDays: 1 },
        ],
      },
      {
        // GD deal — client on CEMIG (Prime + ATMO territory)
        clientId: existingClients[1].id,
        trackType: 'GDL',
        energyType: 'gd_assinatura',
        submarket: 'SE_CO',
        status: 'QUOTES_RECEIVED',
        quotes: [
          { supplierId: primeId, name: 'Prime Energy', price: 268.00, term: 24, structure: 'GD Assinatura BT', responseDays: 1 },
          { supplierId: atmoId, name: 'ATMO Energia', price: 275.50, term: 24, structure: 'GD Assinatura BT', responseDays: 1 },
          { supplierId: delantisId, name: 'Delantis Energias Renováveis', price: 298.00, term: 24, structure: 'GD Compensação Nacional', responseDays: 4 },
          { supplierId: primeId, name: 'Prime Energy', price: 260.00, term: 36, structure: 'GD Assinatura BT', responseDays: 2 },
        ],
      },
      {
        // ACL Convencional — large consumer
        clientId: existingClients[2 % existingClients.length].id,
        trackType: 'ACL',
        energyType: 'Convencional',
        submarket: 'SE_CO',
        status: 'QUOTES_RECEIVED',
        quotes: [
          { supplierId: primeId, name: 'Prime Energy', price: 242.00, term: 24, structure: 'Convencional', responseDays: 2 },
          { supplierId: cemigId, name: 'CEMIG', price: 248.50, term: 24, structure: 'Convencional', responseDays: 1 },
          { supplierId: primeId, name: 'Prime Energy', price: 235.00, term: 36, structure: 'Convencional', responseDays: 2 },
          { supplierId: cemigId, name: 'CEMIG', price: 258.00, term: 12, structure: 'Convencional', responseDays: 3 },
          { supplierId: genialId, name: 'Genial Energia', price: 252.00, term: 24, structure: 'Convencional', responseDays: 5 },
        ],
      },
      {
        // ACL Incentivada — medium consumer
        clientId: existingClients[3 % existingClients.length].id,
        trackType: 'ACL',
        energyType: 'Incentivada 50%',
        submarket: 'S',
        status: 'QUOTES_RECEIVED',
        quotes: [
          { supplierId: primeId, name: 'Prime Energy', price: 218.00, term: 24, structure: 'Incentivada 50%', responseDays: 1 },
          { supplierId: cemigId, name: 'CEMIG', price: 225.00, term: 24, structure: 'Incentivada 50%', responseDays: 2 },
          { supplierId: primeId, name: 'Prime Energy', price: 212.00, term: 36, structure: 'Incentivada 50%', responseDays: 3 },
          { supplierId: atmoId, name: 'ATMO Energia', price: 230.00, term: 12, structure: 'Incentivada 50%', responseDays: 1 },
        ],
      },
    ];

    let dealsCreated = 0;
    let quotesCreated = 0;

    for (const dd of conferenceDeals) {
      const [deal] = await db.insert(deals).values({
        clientId: dd.clientId,
        energyType: dd.energyType,
        submarket: dd.submarket,
        status: dd.status,
        isDemo: true,
        owner: 'admin',
      }).returning();

      // Create track
      await db.insert(dealTracks).values({
        dealId: deal.id,
        type: dd.trackType,
        status: dd.trackType === 'GDL' ? 'GDL_NEW' : 'ACL_NEW',
      });

      // Create quotes using REAL supplier IDs and names
      for (const q of dd.quotes) {
        if (!q.supplierId) continue; // Skip if supplier doesn't exist

        const respondedAt = new Date();
        respondedAt.setDate(respondedAt.getDate() - q.responseDays);
        const createdAt = new Date(respondedAt.getTime() - (q.responseDays * 86400000));

        await db.insert(dealQuotes).values({
          dealId: deal.id,
          supplierId: q.supplierId,
          supplierName: q.name,
          baseEnergyPriceRmwh: String(q.price),
          contractTermMonths: q.term,
          energyType: q.structure,
          status: 'ACTIVE',
          respondedAt,
          createdAt,
        });
        quotesCreated++;
      }
      dealsCreated++;
    }

    res.json({ success: true, results: { deals: dealsCreated, quotes: quotesCreated } });
  } catch (error: any) {
    console.error('[SeedConferenceDeals] Error:', error);
    res.status(500).json({ success: false, error: error.message });
  }
});
