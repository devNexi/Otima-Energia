🎯 PROJECT: Supplier Relationship Manager + Multi-Channel RFO System
The Brazilian Reality Problem:
Each supplier in Brazil has:

Different preferred channels: WhatsApp, Email, Portal, Phone

Different contact persons: João (sales), Maria (quotes), Carlos (commercial)

Different data formats: Some want PDF, some want Excel, some want their portal form

Different response times: Some reply in 2 hours, some in 2 weeks

Current System Gap:
Replit built a great email-only RFO system. But in Brazil, Renan needs to:

Contact Joana at TYR via WhatsApp (she responds fastest there)

Contact Carlos at 2W via their portal (they require portal submissions)

Contact Ana at Comerc via email (she prefers detailed PDF attachments)

Solution: Build a "How Each Supplier Likes to Be Contacted" System

const MultiFormatRFOGenerator = ({ clientId, clientData }) => {
  const [selectedSuppliers, setSelectedSuppliers] = useState([]);
  const [selectedFormat, setSelectedFormat] = useState({});
  const [generatedContent, setGeneratedContent] = useState({});
  
  // Get suppliers with their preferred formats
  const suppliers = [
    {
      id: 1,
      name: 'TYR Energia',
      contacts: [
        { id: 101, name: 'Joana Silva', preferred_format: 'whatsapp', phone: '+55 (11) 99999-9999' }
      ]
    },
    {
      id: 2, 
      name: '2W Energia',
      contacts: [
        { id: 201, name: 'Carlos Mendes', preferred_format: 'portal', format_details: { portal_url: 'https://portal.2w.com.br' } }
      ]
    },
    {
      id: 3,
      name: 'Comerc Energia',
      contacts: [
        { id: 301, name: 'Ana Paula', preferred_format: 'email', email: 'ana@comerc.com.br' }
      ]
    }
  ];

  // When supplier selection changes
  const handleSupplierToggle = (supplierId) => {
    const supplier = suppliers.find(s => s.id === supplierId);
    const primaryContact = supplier.contacts[0];
    
    if (selectedSuppliers.includes(supplierId)) {
      // Remove
      setSelectedSuppliers(selectedSuppliers.filter(id => id !== supplierId));
      const newFormat = {...selectedFormat};
      delete newFormat[supplierId];
      setSelectedFormat(newFormat);
    } else {
      // Add with default format
      setSelectedSuppliers([...selectedSuppliers, supplierId]);
      setSelectedFormat({
        ...selectedFormat,
        [supplierId]: primaryContact.preferred_format
      });
    }
  };

  // Generate content for each selected supplier
  const generateContent = () => {
    const content = {};
    
    selectedSuppliers.forEach(supplierId => {
      const supplier = suppliers.find(s => s.id === supplierId);
      const contact = supplier.contacts[0];
      const format = selectedFormat[supplierId];
      
      // Get template for this format
      const template = getTemplateForFormat(format);
      
      // Fill template with client data
      const filled = fillTemplate(template, {
        client_name: clientData.company_name,
        contact_name: contact.name,
        consumption_kwh: clientData.avg_consumption_kwh,
        // ... other data
      });
      
      content[supplierId] = {
        supplier: supplier.name,
        contact: contact.name,
        format: format,
        content: filled,
        destination: getDestination(contact, format)
      };
    });
    
    setGeneratedContent(content);
  };

  // Get destination string for display
  const getDestination = (contact, format) => {
    switch(format) {
      case 'whatsapp': return `📱 ${contact.phone}`;
      case 'email': return `📧 ${contact.email}`;
      case 'portal': return `🌐 ${contact.format_details?.portal_url}`;
      case 'phone': return `📞 ${contact.phone}`;
      default: return '—';
    }
  };

  // Copy content to clipboard for manual sending
  const copyToClipboard = (supplierId, format) => {
    const content = generatedContent[supplierId].content;
    
    if (format === 'whatsapp') {
      // Copy WhatsApp-ready text
      navigator.clipboard.writeText(content);
      alert('Copiado para WhatsApp! Agora cole no WhatsApp do fornecedor.');
    } else if (format === 'email') {
      // Copy email subject and body
      const emailContent = `Assunto: ${generatedContent[supplierId].subject}\n\n${content}`;
      navigator.clipboard.writeText(emailContent);
      alert('Email copiado! Agora cole no seu cliente de email.');
    }
  };

  return (
    <div className="multi-format-rfo">
      <h4>📤 Solicitar Cotações - Multi-Formato</h4>
      
      {/* SUPPLIER SELECTION WITH FORMATS */}
      <div className="supplier-format-grid">
        {suppliers.map(supplier => {
          const isSelected = selectedSuppliers.includes(supplier.id);
          const primaryContact = supplier.contacts[0];
          
          return (
            <div 
              key={supplier.id}
              className={`supplier-format-card ${isSelected ? 'selected' : ''}`}
            >
              <div className="supplier-checkbox">
                <input
                  type="checkbox"
                  checked={isSelected}
                  onChange={() => handleSupplierToggle(supplier.id)}
                />
              </div>
              
              <div className="supplier-info">
                <div className="supplier-name">{supplier.name}</div>
                <div className="supplier-contact">{primaryContact.name}</div>
              </div>
              
              {/* FORMAT SELECTOR */}
              {isSelected && (
                <div className="format-selector-small">
                  <label>Enviar via:</label>
                  <select
                    value={selectedFormat[supplier.id] || primaryContact.preferred_format}
                    onChange={(e) => setSelectedFormat({
                      ...selectedFormat,
                      [supplier.id]: e.target.value
                    })}
                  >
                    <option value="whatsapp">📱 WhatsApp</option>
                    <option value="email">📧 Email</option>
                    <option value="portal">🌐 Portal</option>
                    <option value="phone">📞 Telefone</option>
                  </select>
                  
                  {/* Format-specific preview */}
                  <div className="format-preview">
                    {selectedFormat[supplier.id] === 'whatsapp' && (
                      <span>Mensagem WhatsApp</span>
                    )}
                    {selectedFormat[supplier.id] === 'portal' && (
                      <a href={primaryContact.format_details?.portal_url} target="_blank">
                        Abrir Portal →
                      </a>
                    )}
                  </div>
                </div>
              )}
            </div>
          );
        })}
      </div>

      {/* GENERATE BUTTON */}
      {selectedSuppliers.length > 0 && (
        <div className="generate-section">
          <button onClick={generateContent} className="btn-generate">
            🧮 Gerar Conteúdo para {selectedSuppliers.length} Fornecedor(es)
          </button>
        </div>
      )}

      {/* GENERATED CONTENT DISPLAY */}
      {Object.keys(generatedContent).length > 0 && (
        <div className="generated-content">
          <h5>📝 Conteúdo Gerado (Copie e Cole)</h5>
          
          {Object.entries(generatedContent).map(([supplierId, content]) => (
            <div key={supplierId} className="content-card">
              <div className="content-header">
                <div className="supplier-title">
                  <strong>{content.supplier}</strong>
                  <span className="format-badge">{content.format}</span>
                </div>
                <div className="content-destination">
                  Para: {content.destination}
                </div>
              </div>
              
              <div className="content-body">
                <pre>{content.content}</pre>
              </div>
              
              <div className="content-actions">
                <button 
                  onClick={() => copyToClipboard(supplierId, content.format)}
                  className="btn-copy"
                >
                  📋 Copiar {content.format === 'whatsapp' ? 'para WhatsApp' : 'para Email'}
                </button>
                
                {content.format === 'portal' && (
                  <a 
                    href={content.destination.replace('🌐 ', '')}
                    target="_blank"
                    className="btn-portal"
                  >
                    🌐 Abrir Portal
                  </a>
                )}
                
                <button className="btn-mark-sent">
                  ✅ Marcar como Enviado
                </button>
              </div>
            </div>
          ))}
          
          {/* BULK ACTIONS */}
          <div className="bulk-actions">
            <button className="btn-secondary">
              📥 Exportar Todos (ZIP)
            </button>
            <button className="btn-primary">
              📊 Criar RFO Tracking
            </button>
          </div>
        </div>
      )}
    </div>
  );
};

// GET /api/templates?format=whatsapp
router.get('/templates', async (req, res) => {
  const { format } = req.query;
  
  const templates = await db.selectFrom('rfo_templates')
    .selectAll()
    .where('is_active', '=', true)
    .where(format ? 'format_type' : '1', '=', format || '1')
    .execute();
  
  res.json(templates);
});

// POST /api/templates/preview
router.post('/templates/preview', async (req, res) => {
  const { templateId, variables } = req.body;
  
  const template = await db.selectFrom('rfo_templates')
    .select(['body_template', 'variables'])
    .where('id', '=', templateId)
    .executeTakeFirst();
  
  // Fill template with variables
  let content = template.body_template;
  Object.entries(variables).forEach(([key, value]) => {
    content = content.replace(new RegExp(`{${key}}`, 'g'), value);
  });
  
  res.json({ content });
});

// GET /api/suppliers/:id/preferences
router.get('/suppliers/:id/preferences', async (req, res) => {
  const supplier = await db.selectFrom('suppliers')
    .leftJoin('supplier_contacts', 'suppliers.id', 'supplier_contacts.supplier_id')
    .select([
      'suppliers.name',
      'supplier_contacts.preferred_format',
      'supplier_contacts.format_details',
      'supplier_contacts.notes'
    ])
    .where('suppliers.id', '=', req.params.id)
    .where('supplier_contacts.is_primary', '=', true)
    .executeTakeFirst();
  
  res.json(supplier);
});
