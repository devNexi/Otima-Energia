Automated Proposal & Contract System for Brazilian Energy Deals
The Problem:
After Renan gets quotes from suppliers (Task 2) and selects the best option, he needs to:

Create a professional proposal to show the client

Generate a formal contract for the client to sign

Calculate exact savings with transparent breakdown

Include all Brazilian regulatory requirements

Make it look like a professional energy consultancy (not a one-person WhatsApp broker)

Current Pain Points:
Renan spends 30-45 minutes creating proposals in Word/Google Docs

Inconsistent formatting between clients

Manual calculations can have errors

No standardized contract templates for Brazil

Hard to track which proposals were sent/rejected

Goal: Build a "Generate Proposta" system that:
Auto-generates beautiful, branded proposals in < 60 seconds

Includes Brazilian energy-specific terms and calculations

Creates ready-to-sign contracts with client/supplier details

Tracks proposal status (sent, viewed, accepted, rejected)

Stores all documents for compliance and audit

🏗️ ARCHITECTURE OVERVIEW
Document Flow:
text
Selected Quote (Task 2)
    → Proposal Generator (Task 4)
    → Creates: 
        1. PDF Proposal (Proposta Comercial)
        2. PDF Contract (Minuta de Contrato)
        3. Savings Summary (Resumo de Economia)
    → Sends to client (email/WhatsApp)
    → Tracks engagement
    → Updates deal status
Key Brazilian Requirements:
Portuguese language with proper energy terminology

Regulatory mentions: ANEEL, CCEE, Lei 15.269/2025

Standard clauses: Modulação, Sazonalidade, Flexibilidade

Legal compliance: LGPD, consumer protection laws

Professional branding: Ótima Energia logo, colors, contact info

CREATE TABLE proposals (
  id SERIAL PRIMARY KEY,
  
  -- Relationships
  client_id INTEGER REFERENCES clients(id) NOT NULL,
  quote_id INTEGER REFERENCES supplier_quotes(id) NOT NULL,
  rfo_id INTEGER REFERENCES rfo_requests(id),
  
  -- Proposal Info
  proposal_number VARCHAR(50) UNIQUE NOT NULL, -- "PROP-2024-001"
  proposal_date DATE NOT NULL DEFAULT CURRENT_DATE,
  valid_until DATE NOT NULL, -- Usually 30 days
  
  -- Document Paths
  proposal_pdf_path VARCHAR(500), -- "/proposals/PROP-2024-001-proposta.pdf"
  contract_pdf_path VARCHAR(500), -- "/proposals/PROP-2024-001-contrato.pdf"
  summary_pdf_path VARCHAR(500), -- "/proposals/PROP-2024-001-resumo.pdf"
  
  -- Financial Details
  current_annual_cost DECIMAL(12,2), -- Client's current cost (R$/ano)
  proposed_annual_cost DECIMAL(12,2), -- New cost with selected quote
  annual_savings DECIMAL(12,2), -- Calculated savings
  savings_percentage DECIMAL(5,2), -- % savings
  
  -- Commission Details
  our_commission_annual DECIMAL(10,2), -- Our yearly commission
  commission_structure VARCHAR(100), -- "R$/MWh", "% savings", etc.
  payment_terms VARCHAR(200), -- "Pagamento mensal", "À vista", etc.
  
  -- Status Tracking
  status VARCHAR(20) DEFAULT 'draft', -- 'draft', 'sent', 'viewed', 'negotiating', 'accepted', 'rejected', 'expired'
  sent_date TIMESTAMP,
  viewed_date TIMESTAMP,
  viewed_count INTEGER DEFAULT 0,
  last_viewed TIMESTAMP,
  
  -- Client Response
  response_date TIMESTAMP,
  response_notes TEXT,
  rejection_reason TEXT,
  
  -- Digital Signature (Future)
  signature_token VARCHAR(100),
  signed_date TIMESTAMP,
  signed_pdf_path VARCHAR(500),
  
  -- Metadata
  created_by VARCHAR(100) DEFAULT 'system',
  ip_address VARCHAR(45), -- For tracking views
  user_agent TEXT,
  
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  
  -- Constraints
  CONSTRAINT valid_dates CHECK (valid_until > proposal_date),
  CONSTRAINT positive_savings CHECK (annual_savings >= 0)
);

-- Indexes
CREATE INDEX idx_proposals_client_id ON proposals(client_id);
CREATE INDEX idx_proposals_quote_id ON proposals(quote_id);
CREATE INDEX idx_proposals_status ON proposals(status);
CREATE INDEX idx_proposals_proposal_number ON proposals(proposal_number);
CREATE INDEX idx_proposals_valid_until ON proposals(valid_until);

🖥️ FRONTEND IMPLEMENTATION

import React, { useState, useEffect } from 'react';
import './ProposalGenerator.css';

const ProposalGenerator = ({ clientId, clientData, selectedQuote }) => {
  const [proposalData, setProposalData] = useState(null);
  const [generating, setGenerating] = useState(false);
  const [previewUrl, setPreviewUrl] = useState(null);
  const [documents, setDocuments] = useState({
    proposal: null,
    contract: null,
    summary: null
  });

  // Initialize with selected quote data
  useEffect(() => {
    if (selectedQuote && clientData) {
      const data = prepareProposalData(clientData, selectedQuote);
      setProposalData(data);
    }
  }, [selectedQuote, clientData]);

  const prepareProposalData = (client, quote) => {
    // Calculate savings
    const currentAnnualCost = client.avg_consumption_kwh * (client.current_price_rmwh / 1000);
    const proposedAnnualCost = quote.total_client_cost_annual;
    const annualSavings = currentAnnualCost - proposedAnnualCost;
    const savingsPercentage = currentAnnualCost > 0 ? (annualSavings / currentAnnualCost) * 100 : 0;

    return {
      // Client Info
      client_name: client.company_name,
      client_cnpj: client.cnpj,
      uc_code: client.last_bill_upload?.uc_code,
      consumption_kwh: client.avg_consumption_kwh?.toLocaleString('pt-BR'),
      demanda_kw: client.avg_demanda_kw?.toLocaleString('pt-BR'),
      distribuidora: client.last_bill_upload?.distribuidora,
      client_city: client.city || 'São Paulo',
      
      // Current Situation
      current_supplier: client.current_supplier || 'Fornecedor atual',
      current_annual_cost: currentAnnualCost.toLocaleString('pt-BR', { minimumFractionDigits: 2 }),
      current_price_rmwh: client.current_price_rmwh?.toLocaleString('pt-BR'),
      
      // Proposal Details
      supplier_name: quote.supplier_name,
      price_structure: quote.price_rmwh 
        ? `R$ ${quote.price_rmwh}/MWh (Preço Fixo)`
        : `PLD + R$ ${quote.pld_spread_rmwh}/MWh`,
      contract_duration: quote.contract_duration,
      contract_type: quote.contract_type,
      contract_start: new Date(quote.contract_start || new Date().setMonth(new Date().getMonth() + 1)).toLocaleDateString('pt-BR'),
      
      // Financials
      proposed_annual_cost: proposedAnnualCost?.toLocaleString('pt-BR', { minimumFractionDigits: 2 }),
      annual_savings: annualSavings.toLocaleString('pt-BR', { minimumFractionDigits: 2 }),
      savings_percentage: savingsPercentage.toFixed(1),
      
      // Commission
      our_commission_annual: quote.our_commission_annual?.toLocaleString('pt-BR', { minimumFractionDigits: 2 }),
      commission_structure: quote.commission_structure || 'R$/MWh',
      commission_paid_by: quote.commission_paid_by === 'supplier' ? 'Fornecedor' : 'Cliente',
      
      // Document Info
      proposal_number: `PROP-${new Date().getFullYear()}${String(new Date().getMonth() + 1).padStart(2, '0')}-${Math.floor(Math.random() * 1000).toString().padStart(3, '0')}`,
      proposal_date: new Date().toLocaleDateString('pt-BR'),
      valid_until: new Date(new Date().setDate(new Date().getDate() + 30)).toLocaleDateString('pt-BR'),
      
      // Company Info
      logo_url: '/logo-otima.png',
      otima_cnpj: '00.000.000/0001-00' // Will be from env/config
    };
  };

  const generateProposal = async () => {
    if (!proposalData) return;
    
    setGenerating(true);
    
    try {
      // Generate PDFs
      const response = await fetch('/api/proposals/generate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          clientId,
          quoteId: selectedQuote.id,
          proposalData
        })
      });
      
      const result = await response.json();
      
      if (result.success) {
        setDocuments({
          proposal: result.documents.proposal,
          contract: result.documents.contract,
          summary: result.documents.summary
        });
        
        // Show preview
        setPreviewUrl(result.documents.proposal.preview_url);
        
        // Create proposal record
        await createProposalRecord(result);
        
        alert('Proposta gerada com sucesso!');
      }
    } catch (error) {
      console.error('Generation error:', error);
      alert('Erro ao gerar proposta.');
    } finally {
      setGenerating(false);
    }
  };

  const createProposalRecord = async (result) => {
    await fetch('/api/proposals', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        client_id: clientId,
        quote_id: selectedQuote.id,
        proposal_number: proposalData.proposal_number,
        proposal_data: proposalData,
        document_paths: {
          proposal: result.documents.proposal.path,
          contract: result.documents.contract.path,
          summary: result.documents.summary.path
        }
      })
    });
  };

  const sendToClient = async (method) => {
    // method: 'email', 'whatsapp', 'download'
    
    if (method === 'email') {
      await sendProposalEmail();
    } else if (method === 'whatsapp') {
      await sendProposalWhatsApp();
    } else {
      // Download all PDFs as ZIP
      window.open(`/api/proposals/${proposalData.proposal_number}/download`, '_blank');
    }
  };

  const sendProposalEmail = async () => {
    const response = await fetch('/api/proposals/send-email', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        clientId,
        proposalNumber: proposalData.proposal_number,
        documents,
        clientEmail: clientData.email
      })
    });
    
    if (response.ok) {
      alert('Proposta enviada por email!');
    }
  };

  const sendProposalWhatsApp = () => {
    const message = `Olá ${clientData.contact_person}, tudo bem?

Envio a proposta comercial da Ótima Energia para análise:

📋 *Proposta:* ${proposalData.proposal_number}
💰 *Economia Anual:* R$ ${proposalData.annual_savings} (${proposalData.savings_percentage}%)
📅 *Válida até:* ${proposalData.valid_until}

Documentos para download:
🔗 ${documents.proposal?.download_url}
🔗 ${documents.contract?.download_url}

Podemos agendar uma reunião para discutir?

Atenciosamente,
Renan - Ótima Energia`;
    
    // Open WhatsApp with pre-filled message
    const whatsappUrl = `https://wa.me/${clientData.phone?.replace(/\D/g, '')}?text=${encodeURIComponent(message)}`;
    window.open(whatsappUrl, '_blank');
  };

  if (!selectedQuote) {
    return (
      <div className="no-quote-selected">
        <p>Selecione uma cotação primeiro para gerar a proposta.</p>
        <p>Vá para a aba "Cotações" e marque uma como selecionada.</p>
      </div>
    );
  }

  return (
    <div className="proposal-generator">
      <h3>📄 Gerar Proposta Comercial</h3>
      
      {/* QUOTE SUMMARY */}
      <div className="quote-summary">
        <div className="summary-header">
          <h4>Cotacao Selecionada</h4>
          <span className="supplier-badge">{selectedQuote.supplier_name}</span>
        </div>
        
        <div className="summary-grid">
          <div className="summary-item">
            <label>Preço</label>
            <span className="value">
              {selectedQuote.price_rmwh 
                ? `R$ ${selectedQuote.price_rmwh}/MWh`
                : `PLD + R$ ${selectedQuote.pld_spread_rmwh}/MWh`
              }
            </span>
          </div>
          
          <div className="summary-item">
            <label>Duração</label>
            <span className="value">{selectedQuote.contract_duration} meses</span>
          </div>
          
          <div className="summary-item">
            <label>Economia Anual</label>
            <span className="value savings">
              R$ {proposalData?.annual_savings} ({proposalData?.savings_percentage}%)
            </span>
          </div>
          
          <div className="summary-item">
            <label>Nossa Comissão</label>
            <span className="value commission">
              R$ {selectedQuote.our_commission_annual?.toLocaleString('pt-BR', { minimumFractionDigits: 2 })}/ano
            </span>
          </div>
        </div>
      </div>

      {/* GENERATE BUTTON */}
      <div className="generate-section">
        <button 
          onClick={generateProposal}
          disabled={generating || !proposalData}

      case 'draft': return '#6a737d';
      case 'sent': return '#0366d6';
      case 'viewed': return '#6f42c1';
      case 'negotiating': return '#dbab09';
      case 'accepted': return '#2ea44f';
      case 'rejected': return '#cf222e';
      case 'expired': return '#6a737d';
      default: return '#6a737d';

  const getDaysRemaining = (validUntil) => {
    const today = new Date();
    const validDate = new Date(validUntil);
    const diffTime = validDate - today;
    return Math.ceil(diffTime / (1000 * 60 * 60 * 24));

      valid_until: new Date(new Date().setDate(new Date().getDate() + 30)),

    }).execute();
    
    res.json({
      success: true,
      documents: {

// Helper: Generate HTML from template
async function generateProposalHTML(data) {
  const template = await db.selectFrom('proposal_templates')
    .select('html_template')
    .where('template_type', '=', 'proposal')
    .where('is_default', '=', true)
    .executeTakeFirst();
  
  let html = template.html_template;
  
  // Replace all variables
  Object.entries(data).forEach(([key, value]) => {
    html = html.replace(new RegExp(`{${key}}`, 'g'), value || '');
  });
  
  // Add CSS styles
  const css = await getProposalCSS();
  html = html.replace('</style>', `${css}</style>`);
  
  return html;
}

// GET /api/proposals/preview/:token
router.get('/preview/:token', async (req, res) => {
  try {
    const { token } = req.params;
    
    // Get proposal by token
    const proposal = await db.selectFrom('proposals')
      .select(['proposal_pdf_path', 'client_id', 'status'])
      .where('signature_token', '=', token)
      .executeTakeFirst();
    
    if (!proposal) {
      return res.status(404).send('Proposta não encontrada');
    }
    
    // Track view
    await db.insertInto('proposal_views').values({
      proposal_id: proposal.id,
      ip_address: req.ip,
      user_agent: req.get('user-agent'),
      view_date: new Date()
    }).execute();
    
    // Update proposal viewed status
    if (proposal.status === 'sent') {
      await db.updateTable('proposals')
        .set({
          status: 'viewed',
          viewed_date: new Date(),
          viewed_count: db.raw('viewed_count + 1'),
          last_viewed: new Date()
        })
        .where('signature_token', '=', token)
        .execute();
    } else {
      await db.updateTable('proposals')
        .set({
          viewed_count: db.raw('viewed_count + 1'),
          last_viewed: new Date()
        })
        .where('signature_token', '=', token)
        .execute();
    }
    
    // Serve PDF
    const filePath = path.join(__dirname, '..', proposal.proposal_pdf_path);
    res.setHeader('Content-Type', 'application/pdf');
    res.setHeader('Content-Disposition', `inline; filename="proposta-${token}.pdf"`);
    res.sendFile(filePath);
    
  } catch (error) {
    console.error('Preview error:', error);
    res.status(500).send('Erro ao carregar proposta');
  }
});
