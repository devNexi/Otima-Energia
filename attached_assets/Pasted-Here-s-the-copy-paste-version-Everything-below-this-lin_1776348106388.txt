**5. Scoring logic (add to the component):**

```typescript
function scoreQuote(q: QuoteRow, allQuotes: QuoteRow[]): {
  totalScore: number;
  priceScore: number;
  slaScore: number;
  termScore: number;
  reasons: string[];
} {
  const prices = allQuotes.map(x => x.baseEnergyPriceRmwh);
  const minPrice = Math.min(...prices);
  const maxPrice = Math.max(...prices);
  const priceRange = maxPrice - minPrice || 1;

  // Price score: 60% weight. Best price = 100, worst = 0.
  const priceScore = ((maxPrice - q.baseEnergyPriceRmwh) / priceRange) * 100;

  // SLA score: 25% weight. Under 2 days = 100, over 5 days = 0.
  const responseDays = q.responseTimeDays ?? 5;
  const slaScore = Math.max(0, Math.min(100, (5 - responseDays) / 3 * 100));

  // Term match score: 15% weight. Prefer 24-36 month terms.
  const termScore = q.termMonths >= 24 && q.termMonths <= 36 ? 100 :
                    q.termMonths === 12 ? 60 :
                    q.termMonths > 36 ? 70 : 50;

  const totalScore = priceScore * 0.6 + slaScore * 0.25 + termScore * 0.15;

  const reasons: string[] = [];
  if (q.baseEnergyPriceRmwh === minPrice) reasons.push('Melhor preГ§o do mercado');
  else if (priceScore > 70) reasons.push(`${((maxPrice - q.baseEnergyPriceRmwh) / maxPrice * 100).toFixed(1)}% abaixo da oferta mais alta`);
  if (responseDays <= 2) reasons.push(`Respondeu em ${responseDays} dia(s) вҖ” dentro do SLA`);
  if (q.termMonths >= 24 && q.termMonths <= 36) reasons.push('Prazo ideal para este perfil');

  return { totalScore, priceScore, slaScore, termScore, reasons };
}
```
