Bill Data Extraction System
Business Context
We are building an energy brokerage in Brazil. Our salesperson (Renan) receives energy bills from clients via WhatsApp/Email as PDFs, JPGs, or PNGs. Currently, he manually types data from these bills into spreadsheets. We need an admin portal feature that:

Accepts bill uploads (PDF/JPG/PNG)

Extracts key data using FREE OCR (Tesseract.js)

Shows extracted data with confidence score

Allows manual correction when OCR fails

Saves structured data to our database

Goal: Reduce bill data entry from 15 minutes to 2 minutes per client.

🏗️ ARCHITECTURE OVERVIEW
Tech Stack
Frontend: React (existing admin portal at /admin)

Backend: Express.js (existing REST API)

Database: PostgreSQL with Drizzle ORM

OCR Engine: Tesseract.js (FREE, no API costs)

PDF Processing: pdf-parse + pdf2image for PDF → image conversion

File Flow:
text
Client Bill (PDF/JPG/PNG) 
    → Upload via Admin Portal
    → Convert to images if PDF
    → OCR with Tesseract.js (Portuguese)
    → Parse Brazilian energy bill patterns
    → Show results with confidence score
    → Manual correction if needed
    → Save to database

CREATE TABLE bill_uploads (
  id SERIAL PRIMARY KEY,
  client_id INTEGER REFERENCES clients(id) NOT NULL,
  
  -- File Info
  file_name VARCHAR(255) NOT NULL,
  file_type VARCHAR(10), -- 'pdf', 'jpg', 'png'
  file_size INTEGER, -- in bytes
  upload_date TIMESTAMP DEFAULT NOW(),
  
  -- OCR Processing
  ocr_raw_text TEXT, -- First 5000 chars of OCR output
  ocr_confidence DECIMAL(3,2), -- 0.00 to 1.00
  ocr_status VARCHAR(20) DEFAULT 'pending', -- 'pending', 'success', 'failed', 'manual'
  
  -- Extracted Data Fields (Brazilian energy bills)
  uc_code VARCHAR(50), -- Unidade Consumidora (e.g., "123.456.789.0")
  consumo_kwh DECIMAL(10,2), -- Consumption in kWh
  demanda_kw DECIMAL(10,2), -- Contracted demand in kW
  valor_total DECIMAL(10,2), -- Total amount in R$
  distribuidora VARCHAR(100), -- Utility company (CPFL, EDP, etc.)
  mes_referencia VARCHAR(20), -- Billing month "MM/AAAA"
  
  -- Metadata
  extraction_method VARCHAR(20), -- 'tesseract', 'manual'
  reviewed_by VARCHAR(100) DEFAULT 'system',
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

import React, { useState } from 'react';
import './BillUploadSection.css';

const BillUploadSection = ({ clientId }) => {
  const [uploading, setUploading] = useState(false);
  const [progress, setProgress] = useState(0);
  const [ocrResult, setOcrResult] = useState(null);
  const [error, setError] = useState(null);

  const handleFileSelect = async (event) => {
    const file = event.target.files[0];
    if (!file) return;
    
    // Validate file type
    const validTypes = ['application/pdf', 'image/jpeg', 'image/png', 'image/jpg'];
    if (!validTypes.includes(file.type)) {
      setError('Tipo de arquivo inválido. Use PDF, JPG ou PNG.');
      return;
    }
    
    // Validate size (10MB max)
    if (file.size > 10 * 1024 * 1024) {
      setError('Arquivo muito grande. Máximo 10MB.');
      return;
    }
    
    setUploading(true);
    setError(null);
    setProgress(10);
    
    try {
      const formData = new FormData();
      formData.append('bill', file);
      
      // Upload and process
      const response = await fetch(`/api/clients/${clientId}/upload-bill`, {
        method: 'POST',
        body: formData,
      });
      
      setProgress(70);
      
      const data = await response.json();
      
      if (data.success) {
        setProgress(100);
        setOcrResult(data);
        
        // Update client list in background
        setTimeout(() => {
          // Refresh client data or update UI
          window.dispatchEvent(new Event('billUploaded'));
        }, 500);
      } else {
        setError(data.error || 'Erro no processamento do arquivo.');
      }
    } catch (err) {
      setError('Erro de conexão. Tente novamente.');
    } finally {
      setUploading(false);
      setTimeout(() => setProgress(0), 1000);
    }
  };

  const handleSaveManual = async (formData) => {
    try {
      const response = await fetch(`/api/bill-uploads/${ocrResult.upload_id}/save-manual`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(formData),
      });
      
      if (response.ok) {
        alert('Dados salvos com sucesso!');
        setOcrResult(null);
        // Trigger page refresh
        window.location.reload();
      }
    } catch (err) {
      alert('Erro ao salvar dados.');
    }
  };

  return (
    <div className="bill-upload-section">
      <h4>📄 Upload de Fatura de Energia</h4>
      
      {error && (
        <div className="error-message">
          ⚠️ {error}
        </div>
      )}
      
      {/* UPLOAD AREA */}
      <div 
        className="upload-area"
        onClick={() => document.getElementById('bill-file-input').click()}
      >
        <div className="upload-icon">📤</div>
        <p className="upload-title">Clique para enviar fatura</p>
        <p className="upload-subtitle">
          Arraste ou clique para enviar PDF, JPG ou PNG
        </p>
        <p className="upload-info">
          <small>Máximo: 10MB • Formatos: PDF, JPG, PNG</small>
        </p>
        <input
          id="bill-file-input"
          type="file"
          accept=".pdf,.jpg,.jpeg,.png"
          style={{ display: 'none' }}
          onChange={handleFileSelect}
          disabled={uploading}
        />
      </div>
      
      {/* UPLOAD PROGRESS */}
      {uploading && (
        <div className="progress-container">
          <div className="progress-bar" style={{ width: `${progress}%` }}></div>
          <div className="progress-text">
            {progress < 50 ? 'Enviando arquivo...' : 'Processando OCR...'} {progress}%
          </div>
        </div>
      )}
      
      {/* OCR RESULTS */}
      {ocrResult && (
        <BillDataForm
          initialData={ocrResult.data}
          confidence={ocrResult.confidence}
          needsManual={ocrResult.needs_manual}
          onSave={handleSaveManual}
          onCancel={() => setOcrResult(null)}
        />
      )}
    </div>
  );
};

export default BillUploadSection;

const express = require('express');
const multer = require('multer');
const Tesseract = require('tesseract.js');
const fs = require('fs').promises;
const path = require('path');
const pdfParse = require('pdf-parse');
const pdf2image = require('pdf2image');

const router = express.Router();
const upload = multer({ 
  storage: multer.memoryStorage(),
  limits: { fileSize: 10 * 1024 * 1024 } // 10MB
});

// POST /api/clients/:id/upload-bill
router.post('/:id/upload-bill', upload.single('bill'), async (req, res) => {
  try {
    const clientId = req.params.id;
    const file = req.file;
    
    if (!file) {
      return res.status(400).json({ 
        success: false, 
        error: 'Nenhum arquivo enviado.' 
      });
    }
    
    console.log(`Processing bill upload for client ${clientId}: ${file.originalname}`);
    
    // Step 1: Save file temporarily
    const tempDir = '/tmp/ocr_uploads';
    await fs.mkdir(tempDir, { recursive: true });
    
    const tempFilePath = path.join(tempDir, `${Date.now()}_${file.originalname}`);
    await fs.writeFile(tempFilePath, file.buffer);
    
    // Step 2: Process based on file type
    let ocrText = '';
    let imagesForOcr = [];
    
    if (file.mimetype === 'application/pdf') {
      // Process PDF: Convert to images
      console.log('Processing PDF file');
      const pdfImages = await pdf2image.convertPDF(tempFilePath, {
        format: 'png',
        singleFile: true,
        outputFolder: tempDir,
        outputFile: `pdf_page_${Date.now()}`,
        page: null // All pages
      });
      
      imagesForOcr = pdfImages.map(img => img.path);
    } else {
      // Process image directly
      console.log('Processing image file');
      imagesForOcr = [tempFilePath];
    }
    
    // Step 3: Run OCR on each image
    console.log(`Running OCR on ${imagesForOcr.length} image(s)`);
    
    for (const imagePath of imagesForOcr) {
      try {
        const result = await Tesseract.recognize(imagePath, 'por', {
          logger: info => {
            if (info.status === 'recognizing text') {
              console.log(`OCR progress: ${info.progress * 100}%`);
            }
          },
          // Optimize for document text
          tessedit_pageseg_mode: '6', // Assume uniform block of text
          tessedit_char_whitelist: '0123456789.,R$kWhWACPFEDNELGM -:/()',
          preserve_interword_spaces: '1'
        });
        
        ocrText += result.data.text + '\n\n';
        console.log(`OCR confidence: ${result.data.confidence}`);
      } catch (ocrError) {
        console.error('OCR error:', ocrError);
        ocrText += '[OCR Error on this page]\n\n';
      }
      
      // Clean up temp image
      await fs.unlink(imagePath).catch(() => {});
    }
    
    // Clean up original file
    await fs.unlink(tempFilePath).catch(() => {});
    
    // Step 4: Parse Brazilian energy bill data
    console.log('Parsing extracted text...');
    const extractedData = parseBrazilianBill(ocrText);
    const confidence = calculateConfidence(extractedData);
    
    console.log(`Extracted data:`, extractedData);
    console.log(`Confidence score: ${confidence}`);
    
    // Step 5: Save to database
    const billUpload = await db.insertInto('bill_uploads').values({
      client_id: clientId,
      file_name: file.originalname,
      file_type: file.mimetype.split('/')[1],
      file_size: file.size,
      ocr_raw_text: ocrText.substring(0, 5000), // Store first 5k chars
      ocr_confidence: confidence,
      ocr_status: confidence >= 0.7 ? 'success' : 'failed',
      uc_code: extractedData.uc,
      consumo_kwh: extractedData.consumo ? parseFloat(extractedData.consumo.replace(/\./g, '').replace(',', '.')) : null,
      demanda_kw: extractedData.demanda ? parseFloat(extractedData.demanda.replace(/\./g, '').replace(',', '.')) : null,
      valor_total: extractedData.valor ? parseFloat(extractedData.valor.replace(/\./g, '').replace(',', '.')) : null,
      distribuidora: extractedData.distribuidora,
      mes_referencia: extractedData.mes,
      extraction_method: 'tesseract',
      reviewed_by: 'system'
    }).returning('id').execute();
    
    const uploadId = billUpload[0].id;
    
    // Step 6: Update client averages
    await updateClientAverages(clientId);
    
    // Step 7: Return results
    res.json({
      success: true,
      upload_id: uploadId,
      data: extractedData,
      confidence: confidence,
      needs_manual: confidence < 0.7,
      message: confidence >= 0.7 
        ? 'Dados extraídos com sucesso! Verifique abaixo.' 
        : 'OCR de baixa confiança. Por favor, insira os dados manualmente.',
      raw_text_preview: ocrText.substring(0, 200) + '...'
    });
    
  } catch (error) {
    console.error('Upload processing error:', error);
    res.status(500).json({
      success: false,
      error: 'Erro no processamento do arquivo. Por favor, insira os dados manualmente.'
    });
  }
});

// Helper function: Parse Brazilian energy bill text
function parseBrazilianBill(text) {
  const upperText = text.toUpperCase().replace(/\s+/g, ' ');
  
  // Enhanced patterns for Brazilian energy bills
  const patterns = {
    // UC Patterns: "UC: 123.456.789.0" or "UNIDADE CONSUMIDORA: 1234567890"
    uc: [
      /UC[\s\:]*([\d\.]+)/,
      /UNIDADE[\s\:]*CONSUMIDORA[\s\:]*([\d\.]+)/,
      /CONTA[\s\:]*CONTRATO[\s\:]*([\d\.]+)/
    ],
    
    // Consumption: "CONSUMO 1.234,56 KWH" or "ENERGIA KWH 1234,56"
    consumo: [
      /CONSUMO[\s\:]*([\d\.,]+)[\s]*K?WH/,
      /ENERGIA[\s\:]*([\d\.,]+)[\s]*K?WH/,
      /KWH[\s\:]*([\d\.,]+)/
    ],
    
    // Demand: "DEMANDA CONTRATADA 123,45 KW" 
    demanda: [
      /DEMANDA[\s\:]*([\d\.,]+)[\s]*K?W/,
      /DEMANDA[\s\:]*CONTRATADA[\s\:]*([\d\.,]+)/,
      /POT[ÊE]NCIA[\s\:]*([\d\.,]+)[\s]*K?W/
    ],
    
    // Total value: "TOTAL A PAGAR R$ 1.234,56" 
    valor: [
      /TOTAL[\s\:A]*PAGAR[\s\:]*R\$\s*([\d\.,]+)/,
      /VALOR[\s\:]*TOTAL[\s\:]*R\$\s*([\d\.,]+)/,
      /R\$\s*([\d\.,]+)[\s]*\(?TOTAL\)?/
    ],
    
    // Distributor: Look for names in text
    distribuidora: [
      /(CPFL[\s\-]*ENERGIA|CPFL)/,
      /(EDP[\s\-]*S[ÃA]O[\s\-]*PAULO|EDP)/,
      /(ENEL[\s\-]*DISTRIBU(I|Ç)[IÍ]|ENEL)/,
      /(CEMIG)/,
      /(LIGHT)/,
      /(NEOENERGIA)/,
      /(ENERGISA)/,
      /(CELESC)/,
      /(CEB[\s\-]*DISTRIBU(I|Ç)[IÍ]|CEB)/
    ],
    
    // Month: "REF 12/2024" or "MÊS/ANO 12/2024"
    mes: [
      /REF[\s\:]*(\d{1,2}\/\d{4})/,
      /M[ÊE]S[\s\/]*ANO[\s\:]*(\d{1,2}\/\d{4})/,
      /PER[IÍ]ODO[\s\:]*(\d{1,2}\/\d{4})/,
      /(\d{2}\/\d{4})[\s]*\(VENCIMENTO\)/
    ]
  };
  
  const result = {};
  
  // Try each pattern group
  for (const [field, patternList] of Object.entries(patterns)) {
    for (const pattern of patternList) {
      const match = upperText.match(pattern);
      if (match && match[1]) {
        result[field] = match[1].trim();
        break; // Use first match found
      }
    }
    
    if (!result[field]) {
      result[field] = null;
    }
  }
  
  return result;
}

// Helper function: Calculate confidence score (0-1)
function calculateConfidence(extractedData) {
  let score = 0;
  let maxScore = 0;
  
  // Field weights (importance)
  const weights = {
    uc: 2.0,      // Most important - uniquely identifies client
    consumo: 1.5,  // Very important - consumption data
    valor: 1.5,    // Very important - financial data
    demanda: 1.0,  // Important for commercial clients
    distribuidora: 0.5, // Nice to have
    mes: 0.5       // Nice to have
  };
  
  // Calculate score
  for (const [field, weight] of Object.entries(weights)) {
    maxScore += weight;
    if (extractedData[field]) {
      score += weight;
    }
  }
  
  // Normalize to 0-1
  const confidence = maxScore > 0 ? score / maxScore : 0;
  
  // Adjust for data quality
  let qualityBonus = 0;
  
  // UC should have dots (Brazilian format)
  if (extractedData.uc && extractedData.uc.includes('.')) {
    qualityBonus += 0.1;
  }
  
  // Consumption should be a reasonable number
  if (extractedData.consumo) {
    const consumoNum = parseFloat(extractedData.consumo.replace(/\./g, '').replace(',', '.'));
    if (consumoNum > 100 && consumoNum < 1000000) { // Reasonable range
      qualityBonus += 0.1;
    }
  }
  
  return Math.min(confidence + qualityBonus, 1.0);
}

// Helper function: Update client averages
async function updateClientAverages(clientId) {
  try {
    // Get all bills for this client
    const bills = await db.selectFrom('bill_uploads')
      .select(['consumo_kwh', 'demanda_kw', 'created_at'])
      .where('client_id', '=', clientId)
      .where('ocr_status', 'in', ['success', 'manual'])
      .where('consumo_kwh', 'is not', null)
      .orderBy('created_at', 'desc')
      .limit(12) // Last 12 months
      .execute();
    
    if (bills.length > 0) {
      const avgConsumption = bills.reduce((sum, bill) => sum + (bill.consumo_kwh || 0), 0) / bills.length;
      const avgDemanda = bills.reduce((sum, bill) => sum + (bill.demanda_kw || 0), 0) / bills.length;
      
      await db.updateTable('clients')
        .set({
          avg_consumption_kwh: avgConsumption,
          avg_demanda_kw: avgDemanda,
          last_bill_upload_id: bills[0].id,
          last_bill_date: bills[0].created_at
        })
        .where('id', '=', clientId)
        .execute();
    }
  } catch (error) {
    console.error('Error updating client averages:', error);
  }
}

// POST /api/bill-uploads/:id/save-manual
router.post('/:id/save-manual', async (req, res) => {
  try {
    const uploadId = req.params.id;
    const { uc, consumo, demanda, valor, distribuidora, mes } = req.body;
    
    // Validate required fields
    if (!uc || !consumo || !valor) {
      return res.status(400).json({
        success: false,
        error: 'UC, Consumo e Valor Total são obrigatórios.'
      });
    }
    
    // Parse Brazilian number format to decimal
    const parseBrazilianNumber = (str) => {
      if (!str) return null;
      return parseFloat(str.replace(/\./g, '').replace(',', '.'));
    };
    
    // Update the bill upload record
    await db.updateTable('bill_uploads')
      .set({
        uc_code: uc,
        consumo_kwh: parseBrazilianNumber(consumo),
        demanda_kw: parseBrazilianNumber(demanda),
        valor_total: parseBrazilianNumber(valor),
        distribuidora: distribuidora || null,
        mes_referencia: mes || null,
        ocr_status: 'manual',
        extraction_method: 'manual',
        reviewed_by: 'renan',
        updated_at: new Date()
      })
      .where('id', '=', uploadId)
      .execute();
    
    // Get client ID for updating averages
    const billRecord = await db.selectFrom('bill_uploads')
      .select('client_id')
      .where('id', '=', uploadId)
      .executeTakeFirst();
    
    if (billRecord) {
      await updateClientAverages(billRecord.client_id);
    }
    
    res.json({
      success: true,
      message: 'Dados salvos com sucesso!'
    });
    
  } catch (error) {
    console.error('Manual save error:', error);
    res.status(500).json({
      success: false,
      error: 'Erro ao salvar dados.'
    });
  }
});
