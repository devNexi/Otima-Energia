typescriptfunction calculateDossierConfidence(dossier: any): 'LOW' | 'MEDIUM' | 'HIGH' {
  let score = 0;
  const checks = [
    { field: 'legalName', weight: 1 },
    { field: 'cnpj', weight: 1 },
    { field: 'distributor', weight: 1 },
    { field: 'submarket', weight: 1 },
    { field: 'connectionType', weight: 1 },
    { field: 'annualConsumptionMWh', weight: 2 },  // Double weight — critical field
    { field: 'averageMonthlyMWh', weight: 1 },
    { field: 'peakDemandKW', weight: 1 },
    { field: 'tariffClass', weight: 1 },
    { field: 'eligibilityType', weight: 1 },
  ];

  const maxScore = checks.reduce((sum, c) => sum + c.weight, 0); // 11

  for (const check of checks) {
    const val = dossier[check.field];
    if (val !== null && val !== undefined && val !== '' && val !== 'NOT_ELIGIBLE_YET') {
      score += check.weight;
    }
  }

  const ratio = score / maxScore;

  if (ratio >= 0.8) return 'HIGH';    // 9+ out of 11 fields
  if (ratio >= 0.5) return 'MEDIUM';  // 6+ out of 11 fields
  return 'LOW';                        // Less than 6 fields
}

Bug 2 — Mixed kWh / MWh units
What's wrong: The client says some records store consumption in kWh and others in MWh. The platform fields say "MWh" but some values were imported as kWh without converting.
Decision: standardize on kWh for display, MWh for internal storage.
The industry standard for commercial energy in Brazil is:

Bills report in kWh (what the client sees on their fatura)
Dossier/ECOS should store in MWh (what suppliers quote in R$/MWh)
Display should show kWh for consumption (familiar to clients) and MWh for pricing/market context

Fix approach:

Bill import → Dossier: Already fixed in the previous prompt (divide by 1000). Verify this is working.
Existing bad data: Find dossiers where consumption is suspiciously high (stored in kWh but labeled as MWh) and correct them:

typescript// Add a one-time cleanup endpoint
app.post("/api/admin/fix-consumption-units", async (req, res) => {
  if (!await validateDealOsSession(req, res)) return;
  try {
    const { clientDossiers } = await import("@shared/schema");
    const { db } = await import("./db");
    const { gt } = await import("drizzle-orm");

    // Find dossiers where annual consumption > 1000 MWh but peak demand < 100 kW
    // These are almost certainly stored in kWh instead of MWh
    const suspicious = await db.select().from(clientDossiers)
      .where(gt(clientDossiers.annualConsumptionMWh, '1000'));

    let fixed = 0;
    for (const d of suspicious) {
      const annual = parseFloat(String(d.annualConsumptionMWh || 0));
      const demand = parseFloat(String(d.peakDemandKW || 0));

      // If consumption is > 1000 "MWh" but demand is < 100 kW, it's probably kWh
      if (annual > 1000 && demand < 100) {
        const correctedAnnual = annual / 1000;
        const correctedMonthly = correctedAnnual / 12;

        await db.update(clientDossiers).set({
          annualConsumptionMWh: String(correctedAnnual.toFixed(2)),
          averageMonthlyMWh: String(correctedMonthly.toFixed(2)),
          updatedAt: new Date(),
        }).where(eq(clientDossiers.id, d.id));
        fixed++;
      }
    }

    res.json({ success: true, checked: suspicious.length, fixed });
  } catch (error: any) {
    res.status(500).json({ success: false, error: error.message });
  }
});
