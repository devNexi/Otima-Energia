Task 2 - Supplier Quote Input & Comparison Dashboard
🎯 PROJECT: Ótima Energia - Supplier Quote Management System
Business Context
After extracting client bill data (Task 1), Renan needs to:

Get price quotes from energy suppliers (TYR, 2W, Comerc, etc.)

Compare different supplier offers side-by-side

Calculate exact savings for the client

Generate professional proposals to close deals

Current Problem: Renan receives supplier quotes via WhatsApp/Email as unstructured PDFs or spreadsheets. He manually compares them in Excel, which is slow and error-prone.

Goal: Build a dashboard where Renan can input supplier quotes and instantly see:

Which supplier is cheapest for this specific client

Exact annual savings compared to client's current costs

Our commission automatically calculated

Professional comparison ready to show the client

🏗️ ARCHITECTURE OVERVIEW
Relationship Flow:
text
Client Bill Data (Task 1) 
    → Supplier Quotes (Task 2) 
    → Comparison Dashboard 
    → Savings Calculation 
    → Proposal Generation
Core Concept:
Each client can have multiple supplier quotes. The system compares them and shows the best option with exact savings calculations.

1. Create supplier_quotes table:

CREATE TABLE supplier_quotes (
  id SERIAL PRIMARY KEY,
  
  -- Relationships
  client_id INTEGER REFERENCES clients(id) NOT NULL,
  bill_upload_id INTEGER REFERENCES bill_uploads(id), -- Links to specific bill analyzed
  
  -- Supplier Information
  supplier_name VARCHAR(100) NOT NULL, -- "TYR", "2W", "Comerc", etc.
  supplier_contact VARCHAR(255), -- Salesperson name/email
  quote_reference VARCHAR(100), -- Supplier's quote ID
  
  -- Quote Dates
  quote_date DATE NOT NULL DEFAULT CURRENT_DATE,
  valid_until DATE NOT NULL, -- Quote expiration
  
  -- PRICING STRUCTURE (Brazilian ACL Specific)
  -- Option 1: Simple Fixed Price
  price_rmwh DECIMAL(10,2), -- R$/MWh (if fixed price)
  
  -- Option 2: PLD Spread (Most Common)
  pld_spread_rmwh DECIMAL(10,2), -- Spread over PLD in R$/MWh
  
  -- Demand Charges
  demanda_price_rkw_mes DECIMAL(10,2), -- R$/kW/month for demand
  
  -- Contract Details
  contract_start DATE,
  contract_duration INTEGER, -- Months (12, 24, 36)
  contract_type VARCHAR(50), -- "Convencional", "11", "15", "Bior"
  
  -- Modulation (Time-of-Day Rates)
  modulacao_ponta_rmwh DECIMAL(10,2), -- Peak hours price
  modulacao_fora_ponta_rmwh DECIMAL(10,2), -- Off-peak price
  modulacao_reservada_rmwh DECIMAL(10,2), -- Reserved capacity price
  
  -- Seasonality (Sazonalidade)
  sazonalidade_seca_rmwh DECIMAL(10,2), -- Dry season spread
  sazonalidade_umida_rmwh DECIMAL(10,2), -- Wet season spread
  
  -- Flexibility
  flexibilidade_percent DECIMAL(5,2), -- Flexibility percentage
  flexibilidade_penalty_rmwh DECIMAL(10,2), -- Penalty for deviation
  
  -- Commission & Financials
  our_commission_rmwh DECIMAL(10,2), -- Our commission in R$/MWh
  commission_percent DECIMAL(5,2), -- Or percentage of savings
  commission_paid_by VARCHAR(20) DEFAULT 'supplier', -- 'supplier' or 'client'
  
  -- Calculated Fields (Auto-calculated)
  total_client_cost_annual DECIMAL(12,2), -- Total R$/year for client
  our_commission_annual DECIMAL(12,2), -- Our commission R$/year
  client_savings_annual DECIMAL(12,2), -- Vs current costs
  
  -- Status & Metadata
  status VARCHAR(20) DEFAULT 'draft', -- 'draft', 'active', 'won', 'lost', 'expired'
  notes TEXT,
  attachment_url VARCHAR(500), -- Link to original quote PDF/Excel
  
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  
  -- Constraints
  CONSTRAINT valid_price CHECK (
    price_rmwh IS NOT NULL OR pld_spread_rmwh IS NOT NULL
  ),
  CONSTRAINT valid_dates CHECK (valid_until > quote_date)
);

-- Indexes
CREATE INDEX idx_supplier_quotes_client_id ON supplier_quotes(client_id);
CREATE INDEX idx_supplier_quotes_supplier ON supplier_quotes(supplier_name);
CREATE INDEX idx_supplier_quotes_status ON supplier_quotes(status);

import React, { useState, useEffect } from 'react';
import './SupplierQuoteManager.css';

const SupplierQuoteManager = ({ clientId, clientData }) => {
  const [quotes, setQuotes] = useState([]);
  const [showAddForm, setShowAddForm] = useState(false);
  const [selectedQuoteId, setSelectedQuoteId] = useState(null);
  const [comparisonView, setComparisonView] = useState(false);

  // Fetch client's quotes
  useEffect(() => {
    fetchQuotes();
  }, [clientId]);

  const fetchQuotes = async () => {
    const response = await fetch(`/api/clients/${clientId}/quotes`);
    const data = await response.json();
    setQuotes(data);
    
    // Set selected quote if client has one
    if (data.find(q => q.status === 'won')) {
      setSelectedQuoteId(data.find(q => q.status === 'won').id);
    }
  };

  // Calculate savings for a quote
  const calculateSavings = (quote) => {
    if (!clientData?.avg_consumption_kwh || !quote.total_client_cost_annual) {
      return null;
    }
    
    const currentAnnualCost = clientData.avg_consumption_kwh * (clientData.current_price_rmwh / 1000);
    const savings = currentAnnualCost - quote.total_client_cost_annual;
    
    return {
      amount: savings,
      percent: (savings / currentAnnualCost) * 100
    };
  };

  return (
    <div className="quote-manager">
      {/* HEADER WITH STATS */}
      <div className="quote-header">
        <h3>📊 Gestão de Cotações</h3>
        <div className="quote-stats">
          <div className="stat-card">
            <div className="stat-value">{quotes.length}</div>
            <div className="stat-label">Cotações</div>
          </div>
          <div className="stat-card">
            <div className="stat-value">
              {quotes.filter(q => q.status === 'active').length}
            </div>
            <div className="stat-label">Ativas</div>
          </div>
          <div className="stat-card">
            <div className="stat-value">
              R$ {quotes.reduce((sum, q) => sum + (q.our_commission_annual || 0), 0).toLocaleString('pt-BR')}
            </div>
            <div className="stat-label">Comissão Total/ano</div>
          </div>
        </div>
      </div>

      {/* ACTION BUTTONS */}
      <div className="action-bar">
        <button 
          className="btn-primary"
          onClick={() => setShowAddForm(true)}
        >
          + Nova Cotação
        </button>
        
        {quotes.length >= 2 && (
          <button 
            className="btn-secondary"
            onClick={() => setComparisonView(!comparisonView)}
          >
            {comparisonView ? 'Ocultar Comparação' : 'Comparar Cotações'}
          </button>
        )}
        
        {selectedQuoteId && (
          <button className="btn-success">
            ✅ Cotação Selecionada
          </button>
        )}
      </div>

      {/* COMPARISON VIEW (When enabled) */}
      {comparisonView && quotes.length >= 2 && (
        <ComparisonDashboard 
          quotes={quotes}
          clientData={clientData}
          onSelectQuote={setSelectedQuoteId}
        />
      )}

      {/* QUOTES LIST TABLE */}
      <div className="quotes-table-container">
        <table className="quotes-table">
          <thead>
            <tr>
              <th>Fornecedor</th>
              <th>Preço (R$/MWh)</th>
              <th>Validade</th>
              <th>Custo Anual</th>
              <th>Economia</th>
              <th>Comissão</th>
              <th>Status</th>
              <th>Ações</th>
            </tr>
          </thead>
          <tbody>
            {quotes.map(quote => (
              <QuoteRow 
                key={quote.id}
                quote={quote}
                clientData={clientData}
                isSelected={selectedQuoteId === quote.id}
                onSelect={() => setSelectedQuoteId(quote.id)}
                onEdit={() => {/* Open edit modal */}}
                onDelete={() => {/* Confirm delete */}}
              />
            ))}
          </tbody>
        </table>
      </div>

      {/* ADD/EDIT QUOTE MODAL */}
      {showAddForm && (
        <QuoteFormModal
          clientId={clientId}
          clientData={clientData}
          onClose={() => setShowAddForm(false)}
          onSave={(newQuote) => {
            setQuotes([...quotes, newQuote]);
            setShowAddForm(false);
          }}
        />
      )}
    </div>
  );
};

export default SupplierQuoteManager;

import React, { useState } from 'react';

const QuoteFormModal = ({ clientId, clientData, onClose, onSave }) => {
  const [formData, setFormData] = useState({
    supplier_name: '',
    quote_reference: '',
    valid_until: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000).toISOString().split('T')[0], // 7 days
    price_type: 'pld_spread', // 'fixed' or 'pld_spread'
    pld_spread_rmwh: '',
    price_rmwh: '',
    demanda_price_rkw_mes: '',
    contract_duration: 12,
    contract_type: 'Convencional',
    our_commission_rmwh: '5.00', // Default R$5/MWh
    commission_paid_by: 'supplier',
    notes: ''
  });

  const [calculating, setCalculating] = useState(false);
  const [calculatedResult, setCalculatedResult] = useState(null);

  // List of common suppliers
  const supplierList = [
    'TYR Energia', '2W Energia', 'Comerc Energia', 'CPFL Energia',
    'EDP Energia', 'AES Brasil', 'Eneva', 'TR Soluções',
    'Blue Soluções', 'Ecogen', 'Outro'
  ];

  // Calculate quote economics
  const calculateQuote = async () => {
    if (!clientData?.avg_consumption_kwh) {
      alert('Primeiro extraia dados da fatura do cliente.');
      return;
    }

    setCalculating(true);
    
    try {
      const response = await fetch('/api/quotes/calculate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          client_data: clientData,
          quote_data: formData
        })
      });
      
      const result = await response.json();
      setCalculatedResult(result);
      
    } catch (error) {
      console.error('Calculation error:', error);
      alert('Erro no cálculo');
    } finally {
      setCalculating(false);
    }
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    
    if (!calculatedResult) {
      alert('Calcule a cotação primeiro para ver os resultados.');
      return;
    }

    const response = await fetch(`/api/clients/${clientId}/quotes`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        ...formData,
        ...calculatedResult
      })
    });

    const savedQuote = await response.json();
    onSave(savedQuote);
  };

  return (
    <div className="modal-overlay">
      <div className="modal-content quote-modal">
        <div className="modal-header">
          <h4>➕ Nova Cotação de Energia</h4>
          <button onClick={onClose} className="close-btn">×</button>
        </div>

        <form onSubmit={handleSubmit}>
          <div className="form-sections">
            
            {/* SECTION 1: SUPPLIER INFO */}
            <div className="form-section">
              <h5>1. Informações do Fornecedor</h5>
              <div className="form-row">
                <div className="form-group">
                  <label>Fornecedor *</label>
                  <select
                    value={formData.supplier_name}
                    onChange={(e) => setFormData({...formData, supplier_name: e.target.value})}
                    required
                  >
                    <option value="">Selecione...</option>
                    {supplierList.map(supplier => (
                      <option key={supplier} value={supplier}>{supplier}</option>
                    ))}
                  </select>
                </div>
                
                <div className="form-group">
                  <label>Referência da Cotação</label>
                  <input
                    type="text"
                    value={formData.quote_reference}
                    onChange={(e) => setFormData({...formData, quote_reference: e.target.value})}
                    placeholder="Ex: COT-2024-001"
                  />
                </div>
                
                <div className="form-group">
                  <label>Validade até *</label>
                  <input
                    type="date"
                    value={formData.valid_until}
                    onChange={(e) => setFormData({...formData, valid_until: e.target.value})}
                    required
                    min={new Date().toISOString().split('T')[0]}
                  />
                </div>
              </div>
            </div>

            {/* SECTION 2: PRICING */}
            <div className="form-section">
              <h5>2. Estrutura de Preços</h5>
              
              <div className="pricing-type-toggle">
                <label>
                  <input
                    type="radio"
                    name="price_type"
                    value="pld_spread"
                    checked={formData.price_type === 'pld_spread'}
                    onChange={(e) => setFormData({...formData, price_type: e.target.value})}
                  />
                  Spread sobre PLD (Comum no Brasil)
                </label>
                
                <label>
                  <input
                    type="radio"
                    name="price_type"
                    value="fixed"
                    checked={formData.price_type === 'fixed'}
                    onChange={(e) => setFormData({...formData, price_type: e.target.value})}
                  />
                  Preço Fixo (R$/MWh)
                </label>
              </div>
              
              <div className="form-row">
                {formData.price_type === 'pld_spread' ? (
                  <div className="form-group">
                    <label>Spread sobre PLD (R$/MWh) *</label>
                    <input
                      type="number"
                      step="0.01"
                      value={formData.pld_spread_rmwh}
                      onChange={(e) => setFormData({...formData, pld_spread_rmwh: e.target.value})}
                      placeholder="Ex: 12.50"
                      required
                    />
                    <small>Preço final = PLD + este spread</small>
                  </div>
                ) : (
                  <div className="form-group">
                    <label>Preço Fixo (R$/MWh) *</label>
                    <input
                      type="number"
                      step="0.01"
                      value={formData.price_rmwh}
                      onChange={(e) => setFormData({...formData, price_rmwh: e.target.value})}
                      placeholder="Ex: 245.00"
                      required
                    />
                  </div>
                )}
                
                <div className="form-group">
                  <label>Demanda (R$/kW/mês)</label>
                  <input
                    type="number"
                    step="0.01"
                    value={formData.demanda_price_rkw_mes}
                    onChange={(e) => setFormData({...formData, demanda_price_rkw_mes: e.target.value})}
                    placeholder="Ex: 15.00"
                  />
                </div>
              </div>
              
              <div className="form-row">
                <div className="form-group">
                  <label>Duração do Contrato (meses) *</label>
                  <select
                    value={formData.contract_duration}
                    onChange={(e) => setFormData({...formData, contract_duration: parseInt(e.target.value)})}
                    required
                  >
                    <option value="12">12 meses</option>
                    <option value="24">24 meses</option>
                    <option value="36">36 meses</option>
                    <option value="48">48 meses</option>
                  </select>
                </div>
                
                <div className="form-group">
                  <label>Tipo de Contrato</label>
                  <select
                    value={formData.contract_type}
                    onChange={(e) => setFormData({...formData, contract_type: e.target.value})}
                  >
                    <option value="Convencional">Convencional</option>
                    <option value="11">Tipo 11</option>
                    <option value="15">Tipo 15</option>
                    <option value="Bior">Bior</option>
                  </select>
                </div>
              </div>
            </div>

            {/* SECTION 3: COMMISSION */}
            <div className="form-section">
              <h5>3. Comissão Ótima Energia</h5>
              <div className="form-row">
                <div className="form-group">
                  <label>Nossa Comissão (R$/MWh) *</label>
                  <input
                    type="number"
                    step="0.01"
                    value={formData.our_commission_rmwh}
                    onChange={(e) => setFormData({...formData, our_commission_rmwh: e.target.value})}
                    placeholder="Ex: 5.00"
                    required
                  />
                  <small>Valor que recebemos por MWh vendido</small>
                </div>
                
                <div className="form-group">
                  <label>Comissão paga por</label>
                  <select
                    value={formData.commission_paid_by}
                    onChange={(e) => setFormData({...formData, commission_paid_by: e.target.value})}
                  >
                    <option value="supplier">Fornecedor (mais comum)</option>
                    <option value="client">Cliente</option>
                  </select>
                </div>
              </div>
            </div>

            {/* CALCULATE BUTTON */}
            <div className="calculate-section">
              <button
                type="button"
                onClick={calculateQuote}
                className="btn-calculate"
                disabled={calculating || !formData.supplier_name}
              >
                {calculating ? '🔄 Calculando...' : '🧮 Calcular Cotação'}
              </button>
            </div>

            {/* CALCULATION RESULTS */}
            {calculatedResult && (
              <div className="results-section">
                <h5>📊 Resultados da Cotação</h5>
                <div className="results-grid">
                  <div className="result-card">
                    <div className="result-label">Custo Anual do Cliente</div>
                    <div className="result-value">
                      R$ {calculatedResult.total_client_cost_annual?.toLocaleString('pt-BR', {minimumFractionDigits: 2})}

                  </div>
                  
                  <div className="result-card">
                    <div className="result-label">Nossa Comissão Anual</div>
                    <div className="result-value success">
                      R$ {calculatedResult.our_commission_annual?.toLocaleString('pt-BR', {minimumFractionDigits: 2})}

                        <span className="savings-negative">

import React from 'react';

const ComparisonDashboard = ({ quotes, clientData, onSelectQuote }) => {
  // Sort quotes by total cost (cheapest first)
  const sortedQuotes = [...quotes].sort((a, b) => 
    (a.total_client_cost_annual || 0) - (b.total_client_cost_annual || 0)
  );

  const currentAnnualCost = clientData?.avg_consumption_kwh 
    ? clientData.avg_consumption_kwh * (clientData.current_price_rmwh / 1000)
    : 0;

  return (
    <div className="comparison-dashboard">
      <h5>📈 Comparação de Cotações</h5>
      
      <div className="comparison-grid">
        {/* CURRENT SITUATION */}
        <div className="comparison-card current">
          <div className="card-header">
            <h6>Situação Atual</h6>
            <div className="card-badge">BASE</div>
          </div>
          <div className="card-body">
            <div className="metric">
              <span className="metric-label">Fornecedor:</span>
              <span className="metric-value">{clientData.current_supplier || 'Não informado'}</span>
            </div>
            <div className="metric">
              <span className="metric-label">Preço:</span>
              <span className="metric-value">
                R$ {clientData.current_price_rmwh?.toLocaleString('pt-BR') || '?'}/MWh
              </span>
            </div>
            <div className="metric">
              <span className="metric-label">Custo Anual:</span>
              <span className="metric-value highlight">
                R$ {currentAnnualCost.toLocaleString('pt-BR', {minimumFractionDigits: 2})}
              </span>
            </div>
            <div className="metric">
              <span className="metric-label">Economia:</span>
              <span className="metric-value">—</span>
            </div>
          </div>
        </div>

        {/* EACH QUOTE */}
        {sortedQuotes.map((quote, index) => {
          const savings = currentAnnualCost - (quote.total_client_cost_annual || 0);
          const savingsPercent = currentAnnualCost > 0 ? (savings / currentAnnualCost) * 100 : 0;
          const isBest = index === 0 && savings > 0;
          
          return (
            <div 
              key={quote.id} 
              className={`comparison-card ${isBest ? 'best' : ''}`}
              onClick={() => onSelectQuote(quote.id)}
            >
              {isBest && <div className="best-badge">⭐ MELHOR</div>}
              
              <div className="card-header">
                <h6>{quote.supplier_name}</h6>
                <div className="card-status">{quote.status}</div>
              </div>
              
              <div className="card-body">
                <div className="metric">
                  <span className="metric-label">Preço:</span>
                  <span className="metric-value">
                    {quote.price_rmwh 
                      ? `R$ ${quote.price_rmwh}/MWh`
                      : `PLD + R$ ${quote.pld_spread_rmwh}/MWh`
                    }
                  </span>
                </div>
                
                <div className="metric">
                  <span className="metric-label">Custo Anual:</span>
                  <span className="metric-value">
                    R$ {quote.total_client_cost_annual?.toLocaleString('pt-BR', {minimumFractionDigits: 2})}
                  </span>
                </div>
                
                <div className="metric">
                  <span className="metric-label">Economia:</span>
                  <span className={`metric-value ${savings > 0 ? 'savings-positive' : 'savings-negative'}`}>
                    {savings > 0 ? '↓ ' : '↑ '}
                    R$ {Math.abs(savings).toLocaleString('pt-BR', {minimumFractionDigits: 2})}
                    {savingsPercent !== 0 && ` (${Math.abs(savingsPercent).toFixed(1)}%)`}
                  </span>
                </div>
                
                <div className="metric">
                  <span className="metric-label">Nossa Comissão:</span>
                  <span className="metric-value commission">
                    R$ {quote.our_commission_annual?.toLocaleString('pt-BR', {minimumFractionDigits: 2})}/ano
                  </span>
                </div>
              </div>
              
              <div className="card-footer">
                <small>Válido até: {new Date(quote.valid_until).toLocaleDateString('pt-BR')}</small>
                {isBest && (
                  <button className="btn-select" onClick={(e) => {
                    e.stopPropagation();
                    onSelectQuote(quote.id);
                  }}>
                    Selecionar
                  </button>
                )}
              </div>
            </div>
          );
        })}
      </div>
      
      {/* SUMMARY STATS */}
      {sortedQuotes.length > 0 && (
